Demonstrates

SQL - Preprocessing data.
The original dataset contains 30.000 rows. Before the grid is created an sql-statements is processed to group them on unique names.
36 records remain afterwards. Any SQL statements allowed by the alasql library can be used.

Show code 

function configureView(viewConfig) {
   var configName = viewConfig.configName;
   if (configName === 'Example28.config') {
      viewConfig.addDataStore('standard-data.json');
      viewConfig.preprocessData = function () {
         // group by name
         var columnInfoGroupBy = ViewHandler.getColumnInfoFromTitle('name');
         var sql = ViewHandler.getSQLSelect();
         sql += ' FROM ' + viewConfig.dataStores[0].alias + ' GROUP BY [' + columnInfoGroupBy.getTitle() + ']';
         return sql;
      }
   }
}
