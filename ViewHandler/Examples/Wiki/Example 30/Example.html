Demonstrates

Using multiple viewhandlers.
A div with a unique id is used for each. "viewhandler", and "viewhandler2"

Show code 

document.addEventListener('DOMContentLoaded', function () {
  createView({ containerId: 'viewhandler', configName: 'Example30-1.config' });
  createView({ containerId: 'viewhandler2', configName: 'Example30-2.config' });
});
function configureView(viewConfig) {
  var configName = viewConfig.configName;
  if (configName === 'Example30-1.config') {
    viewConfig.addDataStore('standard-data.json');
    viewConfig.addColumns([
      { title: 'formating', type: COLUMN.DROPDOWN }
    ]);
  }
  else if (configName === 'Example30-2.config') {
    viewConfig.addDataStore('country-data.json');
    viewConfig.addColumns([
      { title: 'country', type: COLUMN.DROPDOWN }
    ]);
  }
}
